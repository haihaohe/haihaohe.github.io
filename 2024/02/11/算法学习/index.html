int quick_sort(int l,int r,int k)
{
    if(l==r) return q[l];//当区间只有一个数时，返回该数
    int x=q[(l+r)/2],i=l-1,j=r+1;
    while(i<j)
    {
        while(q[++i]<x);
        while(q[--j]>x);
        if(i<j) swap(q[i],q[j]);
	}
    int sl=j-l+1;//左边区间长度
    if(k<=sl) return quick_sort(l,j,k);
    return quick_sort(j+1,r,k-sl);
}

2.归并排序

https://www.acwing.com/activity/content/problem/content/821/

主要思路：

  1. 选择分界点

  2. 递归排序

  3. 归并 合二为一

int temp[N];
void merge_sort(int q[],int l, int r)
{
    if(l>=r) return;
    int mid=(l+r)/2;//选择分界点
    merge_sort(q,l,mid),merge_sort(q,mid+1,r);//递归排序左右两边
    //合并
    int k=0,i=l,j=mid+1;
    while(i<=mid && j<=r)
    {
        if(q[i]<=q[j]) temp[k++]=q[i++];
        else temp[k++]=q[j++];
    }
    while(i<=mid) temp[k++]=q[i++];//将多余的数赋值进临时数组
    while(j<=r) temp[k++]=q[j++];
    //将临时数组中的值放入原数组
    for(i=l,j=0; i<=r ; i++,j++) q[i]=temp[j];
}

应用：逆序对https://www.acwing.com/activity/content/problem/content/822/

思路：归并时，当左半边的数大于右半边的数时，从该数开始到中点，这些数都是逆序数

#include<iostream>
using namespace std;
typedef long long LL;
const int N =1e5+10;
int p[N],temp[N];
LL ans=0;
void merge_sort(int p[],int l,int r)
{
	if(l>=r) return;
	int mid=(l+r)/2;
	merge_sort(p,l,mid),merge_sort(p,mid+1,r);
	int k=0,i=l,j=mid+1;
	while(i<=mid && j<=r)
	{
		if(p[i]<=p[j]) temp[k++]=p[i++];
		else
		{
			ans+=mid-i+1;
			temp[k++]=p[j++];
		}
	}
	while(i<=mid) temp[k++]=p[i++];
	while(j<=r) temp[k++]=p[j++];
	for(i=l,j=0; i<=r; i++,j++) p[i]=temp[j];
}
int main()

	cin>>n;
	for(int i=0; i<n; i++) cin>>p[i];

}

3. 二分

整数二分

二分模板一共有两个，分别适用于不同情况。
算法思路：假设目标值在闭区间[l, r]中， 每次将区间长度缩小一半，当l = r时，我们就找到了目标值。

版本1
当我们将区间[l, r]划分成[l, mid]和[mid + 1, r]时，其更新操作是r = mid或者l = mid + 1;，计算mid时不需要加1。

C++ 代码模板：

int bsearch_1(int l, int r)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;
        else l = mid + 1;
    }
    return l;
}

版本2
当我们将区间[l, r]划分成[l, mid - 1]和[mid, r]时，其更新操作是r = mid - 1或者l = mid;，此时为了防止死循环，计算mid时需要加1。

C++ 代码模板：

int bsearch_2(int l, int r)
{
    while (l < r)
    {
        int mid = l + r + 1 >> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}

vector <int> sub(vector<int> &A,vector<int> &B)
{
	int i=0,t=0;
	vector<int> C;
	while(i<A.size())
	{
		t=A[i]-t;
		if(i<B.size()) t-=B[i];
		if(t<0) C.push_back(t+10);
		else
		C.push_back(t);
		if(t<0) t=1;
		else t=0;
		i++;
	}
	//去除前导0 如100-100 得000
	while(C.size()>1 && C.back() == 0) C.pop_back(); 
	return C;
}

5.2 差分

https://www.acwing.com/problem/content/description/799/

差分是前缀和的逆运算，当
$$
a[i]=b[1]+b[2]+b[3]+…+b[i]时
$$
a数组称为b数组的前缀和，b数组称为a数组的差分，即
$$
b[i]=a[i]-a[i-1]
$$
对a数组中的一段连续的数进行操作时，可转化为对b数组中元素的操作。如：将a数组中第l~r的数字加1。
$$
b[l]+=1,b[r+1]-=1
$$
即可完成操作，这时，
$$
i<=l,a数组中的值加1，i>=r+1,a数组中的值减一，将不该加的抵消
$$

#include<iostream>
using namespace std;
const int N=100010;
int a[N],b[N];
void insert(int l,int r,int c)
{
	b[l]+=c;
	b[r+1]-=c;
}
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1; i<=n ; i++)
	{
		cin>>a[i];
		b[i]=a[i]-a[i-1];
	}
//	for(int i=1; i<=n ; i++) insert(i,i,a[i]);
	while(m--)
	{
		int l,r,c;
		scanf("%d%d%d",&l,&r,&c);
		insert(l,r,c);
	}
	//当b数组插入完成后 求出前缀和即为结果
	 for(int i=1; i<=n ; i++) b[i]+=b[i-1];
	 for(int i=1; i<=n ; i++) printf("%d ",b[i]);
	 cout<<endl;
	 return 0;
}

7.1 n的二进制表示中第k位是几

  1. 先把第k位移到最后一位 n>>k
  2. 看个位是几 x&&1

**>>**是右移操作符，如:

11的二进制形式为：0000 0000 0000 0000 0000 0000 0000 1011，然后把低位的最后两个数字移出，因为该数字是正数，所以在高位补零。则得到的最终结果是0000 0000 0000 0000 0000 0000 0000 0010。转换为十进制是2。

数学意义：右移一位相当于除2，右移n位相当于除以2的n次方。

运用：输出10的二进制表示

#include<iostream>
using namespace std;
int main()
{
	int n=10;
	for(int k=3; k>=0 ; k--) cout<<(n>>k&1);

}

算法：x&-x =x&(~x+1)

~为取反符号

原理：

运用：求出一个数的二进制表示1的个数

#include<iostream>

	return x & -x;

1.3 多重背包问题I

有 N种物品和一个容量是 V的背包。

第 i种物品最多有 si件，每件体积是 vi，价值是 wi。

求解将哪些物品装入背包，可使物品体积总和不超过背包容量，且价值总和最大。
输出最大价值。（数据规模较小 时间复杂度为3次方级别）

思路：转化为01背包 遍历每个物品的件数

#include<bits/stdc++.h>
using namespace std;
const int N =110;
int f[N],v[N],w[N],s[N];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1; i<=n; i++) cin>>v[i]>>w[i]>>s[i];
	for(int i=1; i<=n; i++)
	{
		for(int j=m; j>=v[i]; j--)
		{
			for(int k=1; k<=s[i] && j>=k*v[i]; k++)
				f[j]=max(f[j],f[j-k*v[i]]+k*w[i]);
		}
	}
	cout<<f[m]<<endl;
	return 0;
}
